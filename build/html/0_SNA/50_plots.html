OLS log plot¶

\[\begin{split}\begin{eqnarray} y &=& C\times x^\alpha + m \\ y &\stackrel{\text{i.i.d}}{\sim}& \mathcal{N}(\hat{y}, \sigma^2) \\ \ \\ \log y &=& \alpha \log x+\log C \\ \log y &\not \stackrel{\text{i.i.d}}{\sim}& \mathcal{N}(\hat{\log y}, \sigma^2) \end{eqnarray}\end{split}\]

 1"""
 2========================
 3Visualizing named colors
 4========================
 5Simple plot example with the named colors and its visual representation.
 6"""
 7from __future__ import division
 8
 9import matplotlib.pyplot as plt
10from matplotlib import colors as mcolors
11
12
13colors = dict(mcolors.BASE_COLORS, **mcolors.CSS4_COLORS)
14
15# Sort colors by hue, saturation, value and name.
16by_hsv = sorted((tuple(mcolors.rgb_to_hsv(mcolors.to_rgba(color)[:3])), name)
17                for name, color in colors.items())
18sorted_names = [name for hsv, name in by_hsv]
19
20n = len(sorted_names)
21ncols = 4
22nrows = n // ncols + 1
23
24fig, ax = plt.subplots(figsize=(8, 5))
25
26# Get height and width
27X, Y = fig.get_dpi() * fig.get_size_inches()
28h = Y / (nrows + 1)
29w = X / ncols
30
31for i, name in enumerate(sorted_names):
32    col = i % ncols
33    row = i // ncols
34    y = Y - (row * h) - h
35
36    xi_line = w * (col + 0.05)
37    xf_line = w * (col + 0.25)
38    xi_text = w * (col + 0.3)
39
40    ax.text(xi_text, y, name, fontsize=(h * 0.8),
41            horizontalalignment='left',
42            verticalalignment='center')
43
44    ax.hlines(y + h * 0.1, xi_line, xf_line,
45            color=colors[name], linewidth=(h * 0.6))
46
47ax.set_xlim(0, X)
48ax.set_ylim(0, Y)
49ax.set_axis_off()
